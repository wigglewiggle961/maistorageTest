# Plan 3: Results Report Generator

## Goal
Implement a standalone report generator `scripts/report.py` that reads
`data/eval_results.json` (produced by Plan 2) and generates `data/eval_report.md` — a
human-readable markdown report containing per-question RAGAS scores and aggregate metric
averages for both agentic and baseline systems.

This decouples reporting from the main eval loop, allowing reports to be regenerated
without re-running expensive model inference.

### 2. Report Generation Function

```python
def generate_report(results_path: Path = EVAL_RESULTS_PATH, report_path: Path = EVAL_REPORT_PATH):
    """Read eval_results.json and write eval_report.md."""
    if not results_path.exists():
        print(f"ERROR: {results_path} does not exist. Run scripts/evaluate.py first.")
        sys.exit(1)

    with open(results_path, "r", encoding="utf-8") as f:
        results = json.load(f)

    agent_rows = results.get("agent", [])
    baseline_rows = results.get("baseline", [])
    metrics = ["faithfulness", "answer_relevancy", "context_precision", "context_recall"]

    lines = []
    lines.append("# RAGAS Evaluation Report\n")
    lines.append("> Generated by `scripts/report.py` from `data/eval_results.json`.\n")

    # --- Aggregate Summary Table ---
    lines.append("## Aggregate Scores\n")
    lines.append("| Metric | Baseline | Agent | Delta |")
    lines.append("|--------|----------|-------|-------|")
    for m in metrics:
        baseline_avg = sum(r[m] for r in baseline_rows) / max(len(baseline_rows), 1)
        agent_avg = sum(r[m] for r in agent_rows) / max(len(agent_rows), 1)
        delta = agent_avg - baseline_avg
        arrow = "▲" if delta > 0.001 else ("▼" if delta < -0.001 else "—")
        lines.append(f"| `{m}` | {baseline_avg:.4f} | {agent_avg:.4f} | {arrow} {abs(delta):.4f} |")
    lines.append("")

    # --- Per-Question Breakdown ---
    lines.append("## Per-Question Breakdown\n")
    for i, (a_row, b_row) in enumerate(zip(agent_rows, baseline_rows)):
        lines.append(f"### Q{i + 1}: {a_row['question']}\n")
        lines.append("| Metric | Baseline | Agent |")
        lines.append("|--------|----------|-------|")
        for m in metrics:
            b_score = b_row.get(m, float("nan"))
            a_score = a_row.get(m, float("nan"))
            lines.append(f"| `{m}` | {b_score:.4f} | {a_score:.4f} |")
        lines.append("")
        lines.append(f"**Baseline answer:** {b_row['answer'][:200]}...\n")
        lines.append(f"**Agent answer:** {a_row['answer'][:200]}...\n")

    report_content = "\n".join(lines)
    report_path.parent.mkdir(parents=True, exist_ok=True)
    with open(report_path, "w", encoding="utf-8") as f:
        f.write(report_content)
    print(f"Report written to {report_path}")
```

### 3. Main Entrypoint

    generate_report()
